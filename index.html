  { path: '/about', name: 'About', component: About },

  history: createWebHistory(),

🧪 Code Sample — Login Validation

import users from "@/src/data/users.json";

export default {
  data() {
    return {
      email: "",
      password: "",
    };
  },
  methods: {
    handleLogin() {
      const foundUser = users.find(
        (user) =>
          user.email.toLowerCase() === this.email.trim().toLowerCase() &&
          user.password === this.password
      );

      if (foundUser) {
        localStorage.setItem("user", foundUser.email);
        alert(`Welcome, ${foundUser.username}! You are now logged in.`);
        this.email = "";
        this.password = "";
        this.$router.push("/quiz");
      } else {
        alert("Invalid email or password.");
      }
    },
  },
};

🧪 Code Sample — auth.js Logic

// src/utils/auth.js 
export function isLoggedIn() { 
  return !!localStorage.getItem("user"); 
}
export function logout(router) {
  localStorage.removeItem("user");
  router.push("/");
}

export function listenLoginState(callback) {
  const update = () => callback(isLoggedIn());
  window.addEventListener("storage", update);
  window.addEventListener("user-logged-in", update);
  return () => {
    window.removeEventListener("storage", update);
    window.removeEventListener("user-logged-in", update);
  };
}

// stores/quiz.js
import { defineStore } from 'pinia';

export const useQuizStore = defineStore('quiz', {
  state: () => ({
    result: {
      total: 0,
      answer: 0,
      score: 0,
    },
  }),
  actions: {
    setResult(total, answer, score) {
      this.result = { total, answer, score };
    },
  },
});
